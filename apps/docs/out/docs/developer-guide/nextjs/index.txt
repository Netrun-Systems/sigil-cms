<h2>Quick Start</h2>
<p>Create a catch-all route to render every CMS page:</p>
<pre><code class="language-tsx">// app/[[...slug]]/page.tsx
import { SigilPage, generateSigilMetadata, generateSigilStaticParams } from '@sigil-cms/next';

export async function generateMetadata({ params }: { params: Promise&#x3C;{ slug?: string[] }> }) {
  const { slug } = await params;
  return generateSigilMetadata(slug?.join('/') ?? 'home');
}

export async function generateStaticParams() {
  return generateSigilStaticParams();
}

export default async function Page({ params }: { params: Promise&#x3C;{ slug?: string[] }> }) {
  const { slug } = await params;
  return &#x3C;SigilPage slug={slug?.join('/') ?? 'home'} />;
}
</code></pre>
<p>That's it. Every published page in Sigil is now rendered at its slug.</p>
<h2>Custom Block Components</h2>
<p>Default blocks render clean HTML with <code>sigil-block--*</code> CSS classes. Override any type:</p>
<pre><code class="language-tsx">import { SigilPage } from '@sigil-cms/next';
import type { BlockComponentProps } from '@sigil-cms/next';

function MyHero({ block, content }: BlockComponentProps) {
  return (
    &#x3C;div className="my-hero bg-gradient-to-r from-blue-600 to-purple-600 text-white p-16">
      &#x3C;h1 className="text-5xl font-bold">{content.heading as string}&#x3C;/h1>
      &#x3C;p className="text-xl mt-4">{content.subheading as string}&#x3C;/p>
    &#x3C;/div>
  );
}

export default async function Page({ params }) {
  const { slug } = await params;
  return &#x3C;SigilPage slug={slug?.join('/') ?? 'home'} components={{ hero: MyHero }} />;
}
</code></pre>

<h3>On-demand revalidation</h3>
<pre><code class="language-tsx">// app/api/revalidate/route.ts
import { revalidatePath } from 'next/cache';
import { NextRequest, NextResponse } from 'next/server';

export async function POST(request: NextRequest) {
  const { slug, secret } = await request.json();
  if (secret !== process.env.REVALIDATION_SECRET) {
    return NextResponse.json({ error: 'Invalid secret' }, { status: 401 });
  }
  revalidatePath(`/${slug}`);
  return NextResponse.json({ revalidated: true });
}
</code></pre>
